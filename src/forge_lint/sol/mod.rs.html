107    fn include_lint(&self, lint: SolLint) -> bool {
108        self.severity.as_ref().is_none_or(|sev| sev.contains(&lint.severity()))
109            && self.lints_included.as_ref().is_none_or(|incl| incl.contains(&lint))
110            && !self.lints_excluded.as_ref().is_some_and(|excl| excl.contains(&lint))
111    }
112
113    fn process_source_ast<'ast>(
114        &'ast self,
115        sess: &'ast Session,
116        ast: &'ast ast::SourceUnit<'ast>,
117        file: &SourceFile,
118        path: &Path,
119    ) -> Result<(), diagnostics::ErrorGuaranteed> {
120        // Declare all available passes and lints
121        let mut passes_and_lints = Vec::new();
122        passes_and_lints.extend(high::create_early_lint_passes());
123        passes_and_lints.extend(med::create_early_lint_passes());
124        passes_and_lints.extend(info::create_early_lint_passes());
125
126        // Do not apply 'gas' and 'codesize' severity rules on tests and scripts
127        if !self.path_config.is_test_or_script(path) {
128            passes_and_lints.extend(gas::create_early_lint_passes());
129            passes_and_lints.extend(codesize::create_early_lint_passes());
130        }
131
132        // Filter passes based on linter config
133        let (mut passes, lints): (Vec<Box<dyn EarlyLintPass<'_>>>, Vec<_>) = passes_and_lints
134            .into_iter()
135            .fold((Vec::new(), Vec::new()), |(mut passes, mut ids), (pass, lints)| {
136                let included_ids: Vec<_> = lints
137                    .iter()
138                    .filter_map(|lint| if self.include_lint(*lint) { Some(lint.id) } else { None })
139                    .collect();
140
141                if !included_ids.is_empty() {
142                    passes.push(pass);
143                    ids.extend(included_ids);
144                }
145
146                (passes, ids)
147            });
148
149        // Process the inline-config
150        let comments = Comments::new(file);
151        let inline_config = parse_inline_config(sess, &comments, InlineConfigSource::Ast(ast));
152
153        // Initialize and run the early lint visitor
154        let ctx = LintContext::new(sess, self.with_description, self.config(inline_config), lints);
155        let mut early_visitor = EarlyLintVisitor::new(&ctx, &mut passes);
156        _ = early_visitor.visit_source_unit(ast);
157        early_visitor.post_source_unit(ast);
158
159        Ok(())
160    }
161
162    fn process_source_hir<'hir>(
163        &self,
164        sess: &Session,
165        gcx: &solar_sema::ty::Gcx<'hir>,
166        source_id: hir::SourceId,
167        file: &'hir SourceFile,
168    ) -> Result<(), diagnostics::ErrorGuaranteed> {
169        // Declare all available passes and lints
170        let mut passes_and_lints = Vec::new();
171        passes_and_lints.extend(high::create_late_lint_passes());
172        passes_and_lints.extend(med::create_late_lint_passes());
173        passes_and_lints.extend(info::create_late_lint_passes());
174
175        // Do not apply 'gas' and 'codesize' severity rules on tests and scripts
176        if let FileName::Real(ref path) = file.name
177            && !self.path_config.is_test_or_script(path)
178        {
179            passes_and_lints.extend(gas::create_late_lint_passes());
180            passes_and_lints.extend(codesize::create_late_lint_passes());
181        }
182
183        // Filter passes based on config
184        let (mut passes, lints): (Vec<Box<dyn LateLintPass<'_>>>, Vec<_>) = passes_and_lints
185            .into_iter()
186            .fold((Vec::new(), Vec::new()), |(mut passes, mut ids), (pass, lints)| {
187                let included_ids: Vec<_> = lints
188                    .iter()
189                    .filter_map(|lint| if self.include_lint(*lint) { Some(lint.id) } else { None })
190                    .collect();
191
192                if !included_ids.is_empty() {
193                    passes.push(pass);
194                    ids.extend(included_ids);
195                }
196
197                (passes, ids)
198            });
199
200        // Process the inline-config
201        let comments = Comments::new(file);
202        let inline_config =
203            parse_inline_config(sess, &comments, InlineConfigSource::Hir((&gcx.hir, source_id)));
204
205        // Run late lint visitor
206        let ctx = LintContext::new(sess, self.with_description, self.config(inline_config), lints);
207        let mut late_visitor = LateLintVisitor::new(&ctx, &mut passes, &gcx.hir);
208
209        // Visit this specific source
210        _ = late_visitor.visit_nested_source(source_id);
211
212        Ok(())
213    }
214}

216impl<'a> Linter for SolidityLinter<'a> {
217    type Language = SolcLanguage;
218    type Lint = SolLint;
219
220    /// Build solar session based on the linter config
221    fn init(&self) -> Session {
222        let mut builder = Session::builder();
223        if self.with_json_emitter {
224            let map = Arc::<SourceMap>::default();
225            let json_emitter = JsonEmitter::new(Box::new(std::io::stderr()), map.clone())
226                .rustc_like(true)
227                .ui_testing(false);
228
229            builder = builder.dcx(DiagCtxt::new(Box::new(json_emitter))).source_map(map);
230        } else {
231            builder = builder.with_stderr_emitter();
232        };
233
234        // Create a single session for all files
235        let mut sess = builder.build();
236        sess.dcx = sess.dcx.set_flags(|flags| flags.track_diagnostics = false);
237        sess
238    }
239
240    /// Run AST-based lints.
241    ///
242    /// Note: the `ParsingContext` should already have the sources loaded.
243    fn early_lint<'sess>(&self, input: &[PathBuf], pcx: ParsingContext<'sess>) {
244        let sess = pcx.sess;
245        _ = sess.enter_parallel(|| -> Result<(), diagnostics::ErrorGuaranteed> {
246            // Parse the sources
247            let ast_arena = solar_sema::thread_local::ThreadLocal::new();
248            let ast_result = pcx.parse(&ast_arena);
249
250            // Process each source in parallel
251            ast_result.sources.iter().par_bridge().for_each(|source| {
252                if let (FileName::Real(path), Some(ast)) = (&source.file.name, &source.ast)
253                    && input.iter().any(|input_path| path.ends_with(input_path))
254                {
255                    _ = self.process_source_ast(sess, ast, &source.file, path)
256                }
257            });
258
259            Ok(())
260        });
261    }
262
263    /// Run HIR-based lints.
264    ///
265    /// Note: the `ParsingContext` should already have the sources loaded.
266    fn late_lint<'sess>(&self, input: &[PathBuf], pcx: ParsingContext<'sess>) {
267        let sess = pcx.sess;
268        _ = sess.enter_parallel(|| -> Result<(), diagnostics::ErrorGuaranteed> {
269            // Parse and lower to HIR
270            let hir_arena = solar_sema::thread_local::ThreadLocal::new();
271            let hir_result = pcx.parse_and_lower(&hir_arena);
272
273            if let Ok(Some(gcx_wrapper)) = hir_result {
274                let gcx = gcx_wrapper.get();
275
276                // Process each source in parallel
277                gcx.hir.sources_enumerated().par_bridge().for_each(|(source_id, source)| {
278                    if let FileName::Real(ref path) = source.file.name
279                        && input.iter().any(|input_path| path.ends_with(input_path))
280                    {
281                        _ = self.process_source_hir(sess, &gcx, source_id, &source.file);
282                    }
283                });
284            }
285
286            Ok(())
287        });
288    }
289}

291enum InlineConfigSource<'ast, 'hir> {
292    Ast(&'ast ast::SourceUnit<'ast>),
293    Hir((&'hir hir::Hir<'hir>, hir::SourceId)),
294}
295
296fn parse_inline_config<'ast, 'hir>(
297    sess: &Session,
298    comments: &Comments,
299    source: InlineConfigSource<'ast, 'hir>,
300) -> InlineConfig {
301    let items = comments.iter().filter_map(|comment| {
302        let mut item = comment.lines.first()?.as_str();
303        if let Some(prefix) = comment.prefix() {
304            item = item.strip_prefix(prefix).unwrap_or(item);
305        }
306        if let Some(suffix) = comment.suffix() {
307            item = item.strip_suffix(suffix).unwrap_or(item);
308        }
309        let item = item.trim_start().strip_prefix("forge-lint:")?.trim();
310        let span = comment.span;
311        match InlineConfigItem::parse(item, &ALL_REGISTERED_LINTS) {
312            Ok(item) => Some((span, item)),
313            Err(e) => {
314                sess.dcx.warn(e.to_string()).span(span).emit();
315                None
316            }
317        }
318    });
319
320    match source {
321        InlineConfigSource::Ast(ast) => InlineConfig::from_ast(items, ast, sess.source_map()),
322        InlineConfigSource::Hir((hir, id)) => {
323            InlineConfig::from_hir(items, hir, id, sess.source_map())
324        }
325    }
326}

357impl<'a> TryFrom<&'a str> for SolLint {
358    type Error = SolLintError;
359
360    fn try_from(value: &'a str) -> Result<Self, Self::Error> {
361        for &lint in high::REGISTERED_LINTS {
362            if lint.id() == value {
363                return Ok(lint);
364            }
365        }
366
367        for &lint in med::REGISTERED_LINTS {
368            if lint.id() == value {
369                return Ok(lint);
370            }
371        }
372
373        for &lint in info::REGISTERED_LINTS {
374            if lint.id() == value {
375                return Ok(lint);
376            }
377        }
378
379        for &lint in gas::REGISTERED_LINTS {
380            if lint.id() == value {
381                return Ok(lint);
382            }
383        }
384
385        for &lint in codesize::REGISTERED_LINTS {
386            if lint.id() == value {
387                return Ok(lint);
388            }
389        }
390
391        Err(SolLintError::InvalidId(value.to_string()))
392    }
