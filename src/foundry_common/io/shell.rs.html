49/// Terminal width.
50pub enum TtyWidth {
51    /// Not a terminal, or could not determine size.
52    NoTty,
53    /// A known width.
54    Known(usize),
55    /// A guess at the width.
56    Guess(usize),
57}
58
59impl TtyWidth {
60    /// Returns the width of the terminal from the environment, if known.
61    pub fn get() -> Self {
62        // use stderr
63        #[cfg(unix)]
64        let opt = terminal_size::terminal_size_of(std::io::stderr());
65        #[cfg(not(unix))]
66        let opt = terminal_size::terminal_size();
67        match opt {
68            Some((w, _)) => Self::Known(w.0 as usize),
69            None => Self::NoTty,
70        }
71    }
72
73    /// Returns the width used by progress bars for the tty.
74    pub fn progress_max_width(&self) -> Option<usize> {
75        match *self {
76            Self::NoTty => None,
77            Self::Known(width) | Self::Guess(width) => Some(width),
78        }
79    }
80}

129/// An abstraction around console output that remembers preferences for output
130/// verbosity and color.
131pub struct Shell {
132    /// Wrapper around stdout/stderr. This helps with supporting sending
133    /// output to a memory buffer which is useful for tests.
134    output: ShellOut,
135
136    /// The format to use for message output.
137    output_format: OutputFormat,
138
139    /// The verbosity mode to use for message output.
140    output_mode: OutputMode,
141
142    /// The verbosity level to use for message output.
143    verbosity: Verbosity,
144
145    /// Flag that indicates the current line needs to be cleared before
146    /// printing. Used when a progress bar is currently displayed.
147    needs_clear: AtomicBool,
148}
149
150impl fmt::Debug for Shell {
151    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
152        let mut s = f.debug_struct("Shell");
153        s.field("output_format", &self.output_format);
154        s.field("output_mode", &self.output_mode);
155        s.field("verbosity", &self.verbosity);
156        if let ShellOut::Stream { color_choice, .. } = self.output {
157            s.field("color_choice", &color_choice);
158        }
159        s.finish()
160    }
161}

163/// A `Write`able object, either with or without color support.
164enum ShellOut {
165    /// Color-enabled stdio, with information on whether color should be used.
166    Stream {
167        stdout: AutoStream<std::io::Stdout>,
168        stderr: AutoStream<std::io::Stderr>,
169        stderr_tty: bool,
170        color_choice: ColorChoice,
171    },
172    /// A write object that ignores all output.
173    Empty(std::io::Empty),
174}

176/// Whether messages should use color output.
177#[derive(Debug, Default, PartialEq, Clone, Copy, Serialize, Deserialize, ValueEnum)]
178pub enum ColorChoice {
179    /// Intelligently guess whether to use color output (default).
180    #[default]
181    Auto,
182    /// Force color output.
183    Always,
184    /// Force disable color output.
185    Never,
186}

188impl Default for Shell {
189    fn default() -> Self {
190        Self::new()
191    }

194impl Shell {
195    /// Creates a new shell (color choice and verbosity), defaulting to 'auto' color and verbose
196    /// output.
197    pub fn new() -> Self {
198        Self::new_with(
199            OutputFormat::Text,
200            OutputMode::Normal,
201            ColorChoice::Auto,
202            Verbosity::default(),
203        )
204    }

206    /// Creates a new shell with the given color choice and verbosity.
207    pub fn new_with(
208        format: OutputFormat,
209        mode: OutputMode,
210        color: ColorChoice,
211        verbosity: Verbosity,
212    ) -> Self {
213        Self {
214            output: ShellOut::Stream {
215                stdout: AutoStream::new(std::io::stdout(), color.to_anstream_color_choice()),
216                stderr: AutoStream::new(std::io::stderr(), color.to_anstream_color_choice()),
217                color_choice: color,
218                stderr_tty: std::io::stderr().is_terminal(),
219            },
220            output_format: format,
221            output_mode: mode,
222            verbosity,
223            needs_clear: AtomicBool::new(false),
224        }
225    }

227    /// Creates a shell that ignores all output.
228    pub fn empty() -> Self {
229        Self {
230            output: ShellOut::Empty(std::io::empty()),
231            output_format: OutputFormat::Text,
232            output_mode: OutputMode::Quiet,
233            verbosity: 0,
234            needs_clear: AtomicBool::new(false),
235        }
236    }

240    /// Initializes it with the default values if it has not been set yet.

352    pub fn err(&mut self) -> &mut dyn Write {
353        self.maybe_err_erase_line();
354        self.output.stderr()
355    }

357    /// Erase from cursor to end of line if needed.
358    pub fn maybe_err_erase_line(&mut self) {
359        if self.err_supports_color() && self.set_needs_clear(false) {
360            // This is the "EL - Erase in Line" sequence. It clears from the cursor
361            // to the end of line.
362            // https://en.wikipedia.org/wiki/ANSI_escape_code#CSI_sequences
363            let _ = self.output.stderr().write_all(b"\x1B[K");
364        }
365    }

367    /// Prints a red 'error' message. Use the [`sh_err!`] macro instead.
368    /// This will render a message in [ERROR] style with a bold `Error: ` prefix.
369    ///
370    /// **Note**: will log regardless of the verbosity level.
371    pub fn error(&mut self, message: impl fmt::Display) -> Result<()> {
372        self.maybe_err_erase_line();
373        self.output.message_stderr(&"Error", &ERROR, Some(&message), false)
374    }
375
376    /// Prints an amber 'warning' message. Use the [`sh_warn!`] macro instead.
377    /// This will render a message in [WARN] style with a bold `Warning: `prefix.
378    ///
379    /// **Note**: if `verbosity` is set to `Quiet`, this is a no-op.
380    pub fn warn(&mut self, message: impl fmt::Display) -> Result<()> {
381        match self.output_mode {
382            OutputMode::Quiet => Ok(()),
383            _ => self.print(&"Warning", &WARN, Some(&message), false),
384        }
385    }

413    /// **Note**: if `verbosity` is set to `Quiet`, this is a no-op.
414    pub fn print_err(&mut self, fragment: impl fmt::Display) -> Result<()> {

416            OutputMode::Quiet => Ok(()),
417            _ => self.write_stderr(fragment, &Style::new()),
418        }

421    /// Prints a message, where the status will have `color` color, and can be justified. The
422    /// messages follows without color.
423    fn print(
424        &mut self,
425        status: &dyn fmt::Display,
426        style: &Style,
427        message: Option<&dyn fmt::Display>,
428        justified: bool,
429    ) -> Result<()> {
430        match self.output_mode {
431            OutputMode::Quiet => Ok(()),
432            _ => {
433                self.maybe_err_erase_line();
434                self.output.message_stderr(status, style, message, justified)
435            }
436        }
437    }

440impl ShellOut {
441    /// Prints out a message with a status to stderr. The status comes first, and is bold plus the
442    /// given color. The status can be justified, in which case the max width that will right
443    /// align is 12 chars.
444    fn message_stderr(
445        &mut self,
446        status: &dyn fmt::Display,
447        style: &Style,
448        message: Option<&dyn fmt::Display>,
449        justified: bool,
450    ) -> Result<()> {
451        let buffer = Self::format_message(status, message, style, justified)?;
452        self.stderr().write_all(&buffer)?;
453        Ok(())
454    }
455
456    /// Write a styled fragment
457    fn write_stdout(&mut self, fragment: impl fmt::Display, style: &Style) -> Result<()> {
458        let mut buffer = Vec::new();
459        write!(buffer, "{style}{fragment}{style:#}")?;
460        self.stdout().write_all(&buffer)?;
461        Ok(())
462    }
463
464    /// Write a styled fragment
465    fn write_stderr(&mut self, fragment: impl fmt::Display, style: &Style) -> Result<()> {
466        let mut buffer = Vec::new();
467        write!(buffer, "{style}{fragment}{style:#}")?;
468        self.stderr().write_all(&buffer)?;
469        Ok(())
470    }
471
472    /// Gets stdout as a [`io::Write`](Write) trait object.
473    fn stdout(&mut self) -> &mut dyn Write {
474        match self {
475            Self::Stream { stdout, .. } => stdout,
476            Self::Empty(e) => e,
477        }
478    }
479
480    /// Gets stderr as a [`io::Write`](Write) trait object.
481    fn stderr(&mut self) -> &mut dyn Write {
482        match self {
483            Self::Stream { stderr, .. } => stderr,
484            Self::Empty(e) => e,
485        }
486    }
487
488    /// Formats a message with a status and optional message.
489    fn format_message(
490        status: &dyn fmt::Display,
491        message: Option<&dyn fmt::Display>,
492        style: &Style,
493        justified: bool,
494    ) -> Result<Vec<u8>> {
495        let bold = anstyle::Style::new().bold();
496
497        let mut buffer = Vec::new();
498        if justified {
499            write!(buffer, "{style}{status:>12}{style:#}")?;
500        } else {
501            write!(buffer, "{style}{status}{style:#}{bold}:{bold:#}")?;
502        }
503        match message {
504            Some(message) => {
505                writeln!(buffer, " {message}")?;
506            }
507            None => write!(buffer, " ")?,
508        }
509
510        Ok(buffer)
511    }

514impl ColorChoice {
515    /// Converts our color choice to [`anstream`]'s version.
516    fn to_anstream_color_choice(self) -> anstream::ColorChoice {
517        match self {
518            Self::Always => anstream::ColorChoice::Always,
519            Self::Never => anstream::ColorChoice::Never,
520            Self::Auto => anstream::ColorChoice::Auto,
521        }
522    }
523}
524
525fn supports_color(choice: anstream::ColorChoice) -> bool {
526    match choice {
527        anstream::ColorChoice::Always
528        | anstream::ColorChoice::AlwaysAnsi
529        | anstream::ColorChoice::Auto => true,
530        anstream::ColorChoice::Never => false,
531    }
